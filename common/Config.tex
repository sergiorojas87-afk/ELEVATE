% ============================================================
% Listing styles and callout boxes — accessible color palette
% All colors reference named definitions from the preamble.
% Verified against WCAG AA contrast on background #F8F8F8.
%
% Colorblindness decisions:
%   Keywords  → keywordblue    (#005F9E)  safe for all CVD types
%   Strings   → stringorange   (#D35400)  distinguishable from blue
%                                         by hue AND luminance
%   Comments  → commentgreen   (#3B7A57)  darker than typical green;
%                                         luminance contrast >4.5:1
%   Numbers   → numberpurple   (#5E35B1)  hue orthogonal to all above
%   Frames    → framegray      (#BDBDBD)  neutral, no hue dependency
%   Overflow  → numberpurple   (#5E35B1)  replaced red hook arrow;
%               red+green pairs are the primary CVD failure mode
% ============================================================

% ============================================================
% Callout boxes
%
% Color semantics use shape + label as the primary cue so the
% boxes are never distinguished by hue alone (CVD-safe).
% Each box uses a tinted version of one named color for the
% background fill and the full named color for the frame.
%
% tipbox     → keywordblue   (informational)
% notebox    → stringorange  (attention, non-critical)
% warningbox → functionmagenta (critical / caution)
% infobox    → commentteal   (neutral reference)
% ============================================================
\tcbuselibrary{skins,breakable}

% Tip — informational guidance, blue family
\newtcolorbox{tipbox}[1][]{
  enhanced,
  breakable,
  colback=keywordblue!6!white,
  colframe=keywordblue,
  coltitle=white,
  fonttitle=\bfseries,
  title=Tip,
  #1
}

% Note — requires attention but not critical, orange family
\newtcolorbox{notebox}[1][]{
  enhanced,
  breakable,
  colback=stringorange!8!white,
  colframe=stringorange,
  coltitle=white,
  fonttitle=\bfseries,
  title=Note,
  #1
}

% Warning — critical information, magenta family
% Magenta is chosen over red because:
%   (a) red+green is the most common CVD failure pair and
%       should never be used together as semantic opposites;
%   (b) functionmagenta (#C2185B) has strong luminance contrast
%       against white (#F8F8F8): ratio > 5:1.
\newtcolorbox{warningbox}[1][]{
  enhanced,
  breakable,
  colback=functionmagenta!6!white,
  colframe=functionmagenta,
  coltitle=white,
  fonttitle=\bfseries,
  title=Warning,
  #1
}

% Info — neutral reference material, teal family
\newtcolorbox{infobox}[1][]{
  enhanced,
  breakable,
  colback=commentteal!6!white,
  colframe=commentteal,
  coltitle=white,
  fonttitle=\bfseries,
  title=Information,
  #1
}
